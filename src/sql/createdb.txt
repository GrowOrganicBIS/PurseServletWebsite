DROP TABLE "USERDATA";

CREATE TABLE "USERDATA"
(
"USER_ID" INT not null primary key
        GENERATED ALWAYS AS IDENTITY
        (START WITH 1, INCREMENT BY 1),   
"EMAIL" VARCHAR(50), 
"PASSWORD" VARCHAR(50), 
"FNAME" VARCHAR(50),   
"LNAME" VARCHAR(50),
"USERTYPE" VARCHAR(10)
);

DROP TABLE "INSTRUMENTS";

CREATE TABLE "INSTRUMENTS"
(
"INSTR_ID" INT not null primary key
        GENERATED ALWAYS AS IDENTITY
        (START WITH 1, INCREMENT BY 1),   
"NAME" VARCHAR(50),
"TYPE" VARCHAR(4),
"INTEREST_RATE" FLOAT, 
"VOTING_RIGHTS" CHAR,
"LONG_DESCRIPTION" VARCHAR(100),
"CODE" VARCHAR(50)
);
